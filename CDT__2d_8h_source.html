490  void remove(index_t t) {
491  geo_debug_assert(initialized());
492  if(t == front_) {
493  pop_front();
494  } else if(t == back_) {
495  pop_back();
496  } else {
497  geo_debug_assert(contains(t));
498  index_t t_prev = cdt_.Tprev_[t];
499  index_t t_next = cdt_.Tnext_[t];
500  cdt_.Tprev_[t_next] = t_prev;
501  cdt_.Tnext_[t_prev] = t_next;
502  cdt_.Treset_flag(t,list_id_);
503  }

854  void for_each_T_around_v(
855  index_t v, std::function<bool(index_t t, index_t lv)> doit
856  ) {
857  index_t t = vT(v);
858  index_t lv = index_t(-1);
859  do {
860  lv = Tv_find(t,v);
861  if(doit(t,lv)) {
862  return;
863  }
864  t = Tadj(t, (lv+1)%3);
865  } while(t != vT(v) && t != index_t(-1));
866 
867  // We are done, this was an interior vertex
868  if(t != index_t(-1)) {
869  return;
870  }
871 
872  // It was a vertex on the border, so we need
873  // to traverse the triangle fan in the other
874  // direction until we reach the border again
875  t = vT(v);
876  lv = Tv_find(t,v);
877  t = Tadj(t, (lv+2)%3);
878  while(t != index_t(-1)) {
879  lv = Tv_find(t,v);
880  if(doit(t,lv)) {
881  return;
882  }
883  t = Tadj(t, (lv+2)%3);
884  }
885  }

963  static inline index_t find_3(const index_t* T, index_t v) {
964  // The following expression is 10% faster than using
965  // if() statements. This uses the C++ norm, that
966  // ensures that the 'true' boolean value converted to
967  // an int is always 1. With most compilers, this avoids
968  // generating branching instructions.
969  // Thank to Laurent Alonso for this idea.
970  index_t result = index_t( (T[1] == v) | ((T[2] == v) * 2) );
971  // Sanity check, important if it was T[0], not explicitly
972  // tested (detects input that does not meet the precondition).
973  geo_debug_assert(T[result] == v);
974  return result;
975  }

996  void Tcheck(index_t t) const {
997  if(t == index_t(-1)) {
998  return;
999  }
1000  for(index_t e=0; e<3; ++e) {
1001  geo_assert(Tv(t,e) != Tv(t,(e+1)%3));
1002  if(Tadj(t,e) == index_t(-1)) {
1003  continue;
1004  }
1005  geo_assert(Tadj(t,e) != Tadj(t,(e+1)%3));
1006  index_t t2 = Tadj(t,e);
1007  index_t e2 = Tadj_find(t2,t);
1008  geo_assert(Tadj(t2,e2) == t);
1009  }
1010  }

1139  index_t eT(Edge E) {
1140  index_t v1 = E.first;
1141  index_t v2 = E.second;
1142  index_t result = index_t(-1);
1143  for_each_T_around_v(
