270  intervalRU& operator=(const expansion_nt& rhs) {
271 
272  // Optimized expansion-to-interval conversion:
273  //
274  // Add components starting from the one of largest magnitude
275  // Stop as soon as next component is smaller than ulp (and then
276  // expand interval by ulp).
277 
278  index_t l = rhs.length();
279  ln_ = -rhs.component(l-1);
280  u_ = rhs.component(l-1);
281 
282  for(int comp_idx=int(l)-2; comp_idx>=0; --comp_idx) {
283  double comp = rhs.component(index_t(comp_idx));
284  u_ += comp;
285  ln_ -= comp;
286  if(comp > 0) {
287  double new_u = u_ + comp;
288  if(new_u == u_) {
289  u_ = std::nextafter(
290  u_, std::numeric_limits<double>::infinity()
291  );
292  break;
293  } else {
294  u_ = new_u;
295  }
296  } else {
297  // If we stored l, we would write:
298  // new_l = l + comp
299  // But we store ln = -l, so we write:
300  // -new_ln = -ln + comp
301  // Which means:
302  // new_ln = ln - comp
303 
304  double new_ln = ln_ - comp;
305  if(new_ln == ln_) {
306  ln_ = std::nextafter(
307  ln_, std::numeric_limits<double>::infinity()
308  );
309  break;
310  } else {
311  ln_ = new_ln;
312  }
313  }
314  }
315 
316  control_set(rhs);
317  control_check();
318  return *this;
319  }

323  }

338  Sign2 sign() const {
339  // Branchless
340  int lz = int(ln_ == 0);
341  int ln = int(ln_ > 0); // inverted, it is ln_ !!!
342  int lp = int(ln_ < 0); // inverted, it is ln_ !!!
343  int uz = int(u_ == 0);
344  int un = int(u_ < 0);
345  int up = int(u_ > 0);
346  Sign2 result = Sign2(
347  ln*up*SIGN2_NP+
348  lp*up*SIGN2_PP+
349  lz*up*SIGN2_ZP+
350  ln*un*SIGN2_NN+
351  ln*uz*SIGN2_NZ
352  );
353  result = Sign2(
354  int(result) +
355  int(result==SIGN2_ZERO && !(lz&&uz)) * SIGN2_ERROR
356  );
357  return result;
358  }

386  intervalRU& operator*=(const intervalRU &b) {
387  // get bounds of both operands
388  double aln = ln_;
389  double au = u_;
390  double bln = b.ln_;
391  double bu = b.u_;
392 
393  // negated bounds round to upper
394  // (equivalent to bounds round to lower)
395  double lln = (-aln)*bln;
396  double lun = aln*bu;
397  double uln = au*bln;
398  double uun = (-au)*bu;
399 
400  // bounds round to upper
401  double ll = aln*bln;
402  double lu = (-aln)*bu;
403  double ul = au*(-bln);
404  double uu = au*bu;
405 
406  ln_ = std::max(std::max(lln,lun),std::max(uln,uun));
407  u_ = std::max(std::max(ll,lu),std::max(ul,uu));
408 
409  control_mul(b);
410  control_check();
411  return *this;
412  }

617  intervalRN& operator*=(const intervalRN &x) {

619  double ll = lb_*x.lb_;
620  double lu = lb_*x.ub_;
621  double ul = ub_*x.lb_;
622  double uu = ub_*x.ub_;
623 
624  if(!(ll==ll)) ll = 0.0;
625  if(!(lu==lu)) lu = 0.0;
626  if(!(ul==ul)) ul = 0.0;
627  if(!(uu==uu)) uu = 0.0;
628 
629  if(lu<ll) std::swap(lu, ll);
630  if(ul<ll) std::swap(ul, ll);
631  if(uu<ll) std::swap(uu, ll);
632 
633  if(lu>uu) uu = lu;
634  if(ul>uu) uu = ul;
635 
636  lb_ = ll;
637  ub_ = uu;
638 
639  adjust();
640  } else {
641  lb_ = std::numeric_limits<double>::quiet_NaN();
642  ub_ = std::numeric_limits<double>::quiet_NaN();

651  void adjust() {
652  static constexpr double i = std::numeric_limits<double>::infinity();
653  static constexpr double e = std::numeric_limits<double>::epsilon();
654  // nextafter(1.0) - 1.0
655  static constexpr double m = std::numeric_limits<double>::min();
656  // smallest normalized
657  static constexpr double l = 1.0-e;
658  static constexpr double u = 1.0+e;
659  static constexpr double em = e*m;
660 
661  if(lb_==lb_ && ub_==ub_ && (lb_!=ub_ || (lb_!=i && lb_!=-i))) {
662 
663  if(lb_>ub_) {
664  std::swap(lb_, ub_);
665  }
666 
667  if(lb_>m) {
668  lb_ *= l;
669  } else if(lb_<-m) {
670  lb_ *= u;
671  } else {
672  lb_ -= em;
673  }
674 
675  if(ub_>m) {
676  ub_ *= u;
677  } else if(ub_<-m) {
678  ub_ *= l;
679  } else {
680  ub_ += em;
681  }
682  } else {
683  lb_ = std::numeric_limits<double>::quiet_NaN();
684  ub_ = std::numeric_limits<double>::quiet_NaN();
685  }
686  }
