287  index_t halfedge_vertex(index_t h, index_t dlv) const {
288  index_t f = h/3;
289  index_t lv = (h+dlv)%3;
290  return mesh_.facets.vertex(f,lv);
291  }

293  index_t alpha2(index_t h) const {
294  index_t t1 = h/3;
295  index_t t2 = mesh_.facet_corners.adjacent_facet(h);
296  if(t2 == index_t(-1)) {
297  return index_t(-1);
298  }
299  for(index_t h2: mesh_.facets.corners(t2)) {
300  if(mesh_.facet_corners.adjacent_facet(h2) == t1) {
301  return h2;
302  }
303  }
304  geo_assert_not_reached;
305  }

307  void sew2(index_t h1, index_t h2) {
308  geo_debug_assert(
309  halfedge_vertex(h1,0) == halfedge_vertex(h2,1)
310  );
311  geo_debug_assert(
312  halfedge_vertex(h2,0) == halfedge_vertex(h1,1)
313  );
314  index_t t1 = h1/3;
315  index_t t2 = h2/3;
316  mesh_.facet_corners.set_adjacent_facet(h1,t2);
317  mesh_.facet_corners.set_adjacent_facet(h2,t1);
318  }

350  void save_radial(
351  const std::string& name,
352  vector<index_t>::iterator b, vector<index_t>::iterator e
353  ) {
354  std::ofstream out(name + ".obj");
355  index_t v_ofs = 0;
356  for(vector<index_t>::iterator i=b; i!=e; ++i) {
357  index_t t = (*i/3);
358  index_t v1 = mesh_.facets.vertex(t,0);
359  index_t v2 = mesh_.facets.vertex(t,1);
360  index_t v3 = mesh_.facets.vertex(t,2);
361  out << "v "
362  << vec3(mesh_.vertices.point_ptr(v1)) << std::endl;
363  out << "v "
364  << vec3(mesh_.vertices.point_ptr(v2)) << std::endl;
365  out << "v "

367  out << "f "
368  << v_ofs+1 << " " << v_ofs+2 << " " << v_ofs+3
369  << std::endl;
370  v_ofs += 3;
