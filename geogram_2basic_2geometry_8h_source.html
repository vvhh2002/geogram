285  inline double triangle_area_3d(
286  const double* p1, const double* p2, const double* p3
287  ) {
288  double Ux = p2[0] - p1[0];
289  double Uy = p2[1] - p1[1];
290  double Uz = p2[2] - p1[2];
291 
292  double Vx = p3[0] - p1[0];
293  double Vy = p3[1] - p1[1];
294  double Vz = p3[2] - p1[2];
295 
296  double Nx = Uy*Vz - Uz*Vy;
297  double Ny = Uz*Vx - Ux*Vz;
298  double Nz = Ux*Vy - Uy*Vx;
299  return 0.5 * ::sqrt(Nx*Nx+Ny*Ny+Nz*Nz);
300  }

493  inline void triangle_centroid(
494  const vec3& p, const vec3& q, const vec3& r,
495  double a, double b, double c,
496  vec3& Vg, double& V
497  ) {
498  double abc = a + b + c;
499  double area = Geom::triangle_area(p, q, r);
500  V = area / 3.0 * abc;
501  double wp = a + abc;
502  double wq = b + abc;
503  double wr = c + abc;
504  double s = area / 12.0;
505  Vg.x = s * (wp * p.x + wq * q.x + wr * r.x);
506  Vg.y = s * (wp * p.y + wq * q.y + wr * r.y);
507  Vg.z = s * (wp * p.z + wq * q.z + wr * r.z);
508  }

545  ) {

763  template <class FT> vecng<3,FT> transform_vector(
764  const vecng<3,FT>& v,
765  const Matrix<4,FT>& m
766  ){

769 
770  for(i=0; i<4; i++) {
771  result[i] = 0 ;
772  }

775  result[i] += v[j] * m(j,i) ;

780  result[0], result[1], result[2]
781  ) ;
782  }
