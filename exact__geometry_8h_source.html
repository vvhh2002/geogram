302  inline Sign incircle_2d_SOS(
303  const vec2HE& p0, const vec2HE& p1,
304  const vec2HE& p2, const vec2HE& p3
305  ) {
306  double l0 = (geo_sqr(p0.x) + geo_sqr(p0.y)).estimate() /
307  geo_sqr(p0.w).estimate();
308  double l1 = (geo_sqr(p1.x) + geo_sqr(p1.y)).estimate() /
309  geo_sqr(p1.w).estimate();
310  double l2 = (geo_sqr(p2.x) + geo_sqr(p2.y)).estimate() /
311  geo_sqr(p2.w).estimate();
312  double l3 = (geo_sqr(p3.x) + geo_sqr(p3.y)).estimate() /
313  geo_sqr(p3.w).estimate();
314  return incircle_2d_SOS_with_lengths(p0,p1,p2,p3,l0,l1,l2,l3);
315  }
