282  inline void* aligned_malloc(
283  size_t size, size_t alignment = GEO_MEMORY_ALIGNMENT
284  ) {
285 #if defined(GEO_OS_ANDROID)
286  // Alignment not supported under Android.
287  geo_argused(alignment);
288  return malloc(size);
289 #elif defined(GEO_COMPILER_INTEL)
290  return _mm_malloc(size, alignment);
291 #elif defined(GEO_COMPILER_GCC) || defined(GEO_COMPILER_CLANG)
292  void* result;
293  return posix_memalign(&result, alignment, size) == 0
294  ? result : nullptr;
295 #elif defined(GEO_COMPILER_MSVC)
296  return _aligned_malloc(size, alignment);
297 #else
298  geo_argused(alignment);
299  return malloc(size);
300 #endif
301  }

429  inline void* align(void* p) {
430  size_t offset = (
431  GEO_MEMORY_ALIGNMENT -
432  (reinterpret_cast<size_t>(p) & (GEO_MEMORY_ALIGNMENT - 1))
433  ) & (GEO_MEMORY_ALIGNMENT - 1);
434  return reinterpret_cast<char*>(p) + offset;
435  }
